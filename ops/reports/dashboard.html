// --- Helpers --------------------------------------------------------- function pct(value, decimals = 0) { if (value === undefined || value === null || isNaN(value)) return null; const v = Number(value) * 100; return v.toFixed(decimals); } function formatTs(ts) { if (!ts) return "–"; try { const d = new Date(ts); if (isNaN(d.getTime())) return ts; return d.toISOString().slice(0, 16).replace("T", " "); } catch { return ts; } } function classifyKpi(value) { if (value === null) return { cls: "", label: "n/a" }; const v = Number(value); if (v >= 80) return { cls: "kpi-good", label: "good" }; if (v >= 60) return { cls: "kpi-warn", label: "watch" }; return { cls: "kpi-danger", label: "critical" }; } function parseJsonl(text) { const lines = text.split("\n").map(l => l.trim()).filter(Boolean); const records = []; for (const line of lines) { try { records.push(JSON.parse(line)); } catch (e) { console.warn("Skipping invalid JSONL line", line); } } return records; } function normalizeStatus(raw) { if (!raw) return {}; // Case A: altes "HealthState"-Format (flat) if (raw.agent_response_success_rate !== undefined) return raw; // Case B: neues system_status.json (dein aktuelles) const metrics = raw.health?.metrics || {}; return { // KPIs, die updateStatusCard() erwartet autonomy_level_estimate: raw.autonomy?.current_level ?? null, // 0..1 agent_response_success_rate: metrics.agent_response_success_rate ?? null, // 0..1 system_stability_score: metrics.system_stability_score ?? null, // 0..1 self_detection_errors: metrics.self_detected_errors_24h ?? null, // int // optional (falls du sie später nutzen willst) total_actions: metrics.total_actions ?? null, failed_actions: metrics.failed_actions ?? null, // fürs “last action”/timestamp im Caption-Text last_autonomous_action: raw.autonomy?.last_autonomous_action ?? null, timestamp: raw.generated_at ?? raw.generatedAt ?? raw.timestamp ?? null }; } // --- Chart ----------------------------------------------------------- let autonomyChart; function buildAutonomyChart(ctx, records) { if (!records || !records.length) return; const last40 = records.slice(-40); const labels = last40.map(r => formatTs(r.last_autonomous_action || r.timestamp)); const autonomy = last40.map(r => Number(r.autonomy_level_estimate ?? 0) * 100); const success = last40.map(r => Number(r.agent_response_success_rate ?? 0) * 100); const stability = last40.map(r => Number(r.system_stability_score ?? 0) * 100); const data = { labels, datasets: [ { label: "Autonomy level", data: autonomy, borderColor: "#00e0ff", backgroundColor: "rgba(56,189,248,0.2)", tension: 0.3, fill: true, }, { label: "Success rate", data: success, borderColor: "#22c55e", backgroundColor: "rgba(34,197,94,0.15)", tension: 0.3, fill: false, }, { label: "Stability", data: stability, borderColor: "#facc15", backgroundColor: "rgba(250,204,21,0.15)", tension: 0.3, fill: false, } ] }; const options = { responsive: true, maintainAspectRatio: false, scales: { y: { beginAtZero: true, max: 100, ticks: { color: "#9ca3af", callback: (v) => v + "%" }, grid: { color: "rgba(55,65,81,0.4)" } }, x: { ticks: { color: "#6b7280", maxRotation: 0, autoSkip: true, maxTicksLimit: 6 }, grid: { display: false } } }, plugins: { legend: { labels: { color: "#9ca3af", usePointStyle: true } }, tooltip: { callbacks: { label: (ctx) => `${ctx.dataset.label}: ${ctx.parsed.y.toFixed(1)}%` } } } }; if (autonomyChart) autonomyChart.destroy(); autonomyChart = new Chart(ctx, { type: "line", data, options }); } // --- Rendering ------------------------------------------------------- function updateStatusCard(status) { const autonomyMain = document.getElementById("autonomy-main"); const autonomyCaption = document.getElementById("autonomy-caption"); const kSuccess = document.getElementById("kpi-success"); const kStability = document.getElementById("kpi-stability"); const kErrors = document.getElementById("kpi-errors"); const kTotal = document.getElementById("kpi-total"); const kFailed = document.getElementById("kpi-failed"); const kLastAction = document.getElementById("kpi-last-action"); const successChip = document.getElementById("kpi-success-chip"); const stabilityChip = document.getElementById("kpi-stability-chip"); const errorsChip = document.getElementById("kpi-errors-chip"); const autonomy = pct(status.autonomy_level_estimate ?? 0); const success = pct(status.agent_response_success_rate ?? 0); const stability = pct(status.system_stability_score ?? 0); const selfErrors = status.self_detection_errors ?? 0; const totalActions = status.total_actions ?? 0; const failedActions = status.failed_actions ?? 0; const lastAction = status.last_autonomous_action || status.timestamp; autonomyMain.innerHTML = (autonomy ?? "–") + "%"; const when = status.timestamp || status.last_autonomous_action; autonomyCaption.textContent = when ? `Latest snapshot: ${formatTs(when)}${status.last_autonomous_action ? ` • last action: ${status.last_autonomous_action}` : ""}` : "No snapshot recorded yet."; // KPIs kSuccess.textContent = success !== null ? success + "%" : "–"; kStability.textContent = stability !== null ? stability + "%" : "–"; kErrors.textContent = selfErrors; kTotal.textContent = totalActions; kFailed.textContent = failedActions; kLastAction.textContent = formatTs(lastAction); // Chips [successChip, stabilityChip, errorsChip].forEach(chip => { chip.classList.remove("kpi-good", "kpi-warn", "kpi-danger"); }); const sc = classifyKpi(success); successChip.textContent = sc.label; if (sc.cls) successChip.classList.add(sc.cls); const st = classifyKpi(stability); stabilityChip.textContent = st.label; if (st.cls) stabilityChip.classList.add(st.cls); let errCls = ""; let errLabel = "n/a"; if (selfErrors === 0) { errCls = "kpi-good"; errLabel = "clean"; } else if (selfErrors < 3) { errCls = "kpi-warn"; errLabel = "minor"; } else { errCls = "kpi-danger"; errLabel = "frequent"; } errorsChip.textContent = errLabel; if (errCls) errorsChip.classList.add(errCls); const lastUpdated = document.getElementById("last-updated"); lastUpdated.textContent = "Last updated · " + new Date().toLocaleString(); } function updateHistoryTable(records) { const tbody = document.getElementById("history-table"); tbody.innerHTML = ""; if (!records || !records.length) { tbody.innerHTML = "No health history yet."; return; } const last20 = records.slice(-20).reverse(); for (const r of last20) { const autonomy = pct(r.autonomy_level_estimate ?? 0); const success = pct(r.agent_response_success_rate ?? 0); const stability = pct(r.system_stability_score ?? 0); const errs = r.self_detection_errors ?? 0; const total = r.total_actions ?? 0; const failed = r.failed_actions ?? 0; const ts = formatTs(r.last_autonomous_action || r.timestamp); const tr = document.createElement("tr"); tr.innerHTML = ` ${ts} ${autonomy !== null ? autonomy + "%" : "–"} ${success !== null ? success + "%" : "–"} ${stability !== null ? stability + "%" : "–"} ${errs} ${total} / ${failed} `; tbody.appendChild(tr); } } // --- Main load ------------------------------------------------------- async function loadDashboard() { const autonomyError = document.getElementById("autonomy-error"); const historyError = document.getElementById("history-error"); autonomyError.style.display = "none"; historyError.style.display = "none"; try { const [statusRes, logRes] = await Promise.all([ fetch(STATUS_URL, { cache: "no-store" }), fetch(HEALTH_LOG_URL, { cache: "no-store" }), ]); if (!statusRes.ok) { throw new Error("Failed to fetch status.json: " + statusRes.status); } let statusJson = {}; try { statusJson = await statusRes.json(); } catch (e) { console.warn("Invalid status.json", e); } // status.json might be directly in HealthState or in a separate part const status = normalizeStatus(statusJson); updateStatusCard(status); let records = []; if (logRes.ok) { const text = await logRes.text(); records = parseJsonl(text); } else { console.warn("No health_log.jsonl yet"); } updateHistoryTable(records); const ctx = document.getElementById("autonomyChart").getContext("2d"); if (records.length) { buildAutonomyChart(ctx, records); } else { autonomyError.style.display = "block"; autonomyError.textContent = "No health history yet – run GEORGE orchestrator V2 to generate data."; } } catch (err) { console.error(err); autonomyError.style.display = "block"; autonomyError.textContent = "Failed to load live data from GitHub. Check CORS/URLs or try again later."; historyError.style.display = "block"; historyError.textContent = "No history available due to fetch error."; } } document.addEventListener("DOMContentLoaded", () => { loadDashboard(); // Optional: alle 5 Minuten aktualisieren setInterval(loadDashboard, 5 * 60 * 1000); });
